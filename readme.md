Консольное приоложение для просмотра погоды в разных городах. 

Запуск - python3 main.py {Город} / --долгота --широта

# Получить погоду по названию города
python main.py Москва

# Получить погоду по координатам
python main.py --lat 55.7558 --lon 37.6173

# Принудительное обновление данных (игнорировать кэш)
python main.py Лондон --refresh

# Показать справку
python main.py --help
